


SELECT  SUM(AWBFLIGHTLINK.PCS),SUM(AWBFLIGHTLINK.GROSSWT), FLIGHTDETAILS.FLIGHT,  
FLIGHTDETAILS.FLIGHTDATE,CALLQUEUE.RTIMESTAMP,FLIGHTDETAILS.STDARRIVALTIME,
DRIVER.REGNNO,CALLQUEUE.GATENO,CALLQUEUE.DOOR_ALLOC_TS,JOB.EVENT1  FROM JOB, AWBFLIGHTLINK, FLIGHTDETAILS, CALLQUEUE, DRIVER 
WHERE  JOB.TENANTID=88  AND JOB.PRODUCTTYPE='FLT' AND JOB.TENANTID=FLIGHTDETAILS.TENANTID 
AND  JOB.QUEUEID= CALLQUEUE.QUEUEID  AND  JOB.FLIGHT = FLIGHTDETAILS.FLIGHT  AND JOB.FLIGHTDATE= FLIGHTDETAILS.FLIGHTDATE 
AND DRIVER.DRIVERID=CALLQUEUE.DRIVERID  AND JOB.EVENT3 IS NULL AND AWBFLIGHTLINK.FLIGHTDETAILSID = FLIGHTDETAILS.FLIGHTDETAILSID  
AND VARCHAR_FORMAT(CALLQUEUE.RTIMESTAMP,'DD-MM-YYYY')='19-08-2019' AND JOB.EVENT3 IS NULL  
GROUP BY DRIVER.NAME,FLIGHTDETAILS.FLIGHT, FLIGHTDETAILS.FLIGHTDATE, CALLQUEUE.RTIMESTAMP,FLIGHTDETAILS.STDARRIVALTIME,
DRIVER.REGNNO,CALLQUEUE.GATENO,CALLQUEUE.DOOR_ALLOC_TS,JOB.EVENT1 ;




SELECT SUM(AWBFLIGHTLINK.PCS),SUM(AWBFLIGHTLINK.GROSSWT), FLIGHTDETAILS.FLIGHT, 
FLIGHTDETAILS.FLIGHTDATE,CALLQUEUE.RTIMESTAMP,FLIGHTDETAILS.STDARRIVALTIME,
DRIVER.REGNNO,CALLQUEUE.GATENO,CALLQUEUE.DOOR_ALLOC_TS,JOB.EVENT1 
FROM JOB, AWBFLIGHTLINK, FLIGHTDETAILS, CALLQUEUE, DRIVER 
LEFT JOIN JOB ON JOB.PRODUCTID = AWBFLIGHTLINK.FLIGHT
LEFT JOIN CALLQUEUE ON CALLQUEUE.QUEUEID = JOB.QUEUEID
WHERE JOB.TENANTID=88 AND JOB.PRODUCTTYPE='FLT' AND JOB.TENANTID=FLIGHTDETAILS.TENANTID 
AND JOB.QUEUEID= CALLQUEUE.QUEUEID AND JOB.FLIGHT = FLIGHTDETAILS.FLIGHT AND JOB.FLIGHTDATE= FLIGHTDETAILS.FLIGHTDATE 
AND DRIVER.DRIVERID=CALLQUEUE.DRIVERID AND JOB.EVENT3 IS NULL AND AWBFLIGHTLINK.FLIGHTDETAILSID = FLIGHTDETAILS.FLIGHTDETAILSID 
AND VARCHAR_FORMAT(CALLQUEUE.RTIMESTAMP,'DD-MM-YYYY')='19-08-2019' AND JOB.EVENT3 IS NULL 
GROUP BY DRIVER.NAME,FLIGHTDETAILS.FLIGHT, FLIGHTDETAILS.FLIGHTDATE, CALLQUEUE.RTIMESTAMP,FLIGHTDETAILS.STDARRIVALTIME,
DRIVER.REGNNO,CALLQUEUE.GATENO,CALLQUEUE.DOOR_ALLOC_TS,JOB.EVENT1 ;
